select `order`.ocode, `order`.day, store.sname, `order`.tabnam, `order`.flg, menu.mname, list.quantity
    from `order`, list, store, menu
    where store.scode = `order`.scode and
    `order`.ocode = list.ocode and
    list.mcode = menu.mcode;
